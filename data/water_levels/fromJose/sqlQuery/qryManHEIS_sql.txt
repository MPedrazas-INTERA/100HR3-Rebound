SELECT HEIS2_ADM_HYDRAULIC_HEAD_MV.WELL_NAME AS NAME, HEIS2_ADM_HYDRAULIC_HEAD_MV.HYD_DATE_TIME_PST AS EVENT, HEIS2_ADM_HYDRAULIC_HEAD_MV.HYD_HEAD_METERS_NAVD88 AS VAL, "MAN" AS TYPE, HEIS2_ADM_HORIZONTAL_COORD.EASTING, HEIS2_ADM_HORIZONTAL_COORD.NORTHING
FROM HEIS2_ADM_HYDRAULIC_HEAD_MV INNER JOIN HEIS2_ADM_HORIZONTAL_COORD ON HEIS2_ADM_HYDRAULIC_HEAD_MV.WELL_ID = HEIS2_ADM_HORIZONTAL_COORD.WELL_ID
WHERE (((HEIS2_ADM_HYDRAULIC_HEAD_MV.HYD_DATE_TIME_PST)>=#1/1/2014#) AND ((HEIS2_ADM_HYDRAULIC_HEAD_MV.HYD_HEAD_METERS_NAVD88) Is Not Null) AND ((HEIS2_ADM_HORIZONTAL_COORD.EASTING)>=571515.1228 And (HEIS2_ADM_HORIZONTAL_COORD.EASTING)<=579316.4375) AND ((HEIS2_ADM_HORIZONTAL_COORD.NORTHING)>=150216.660799999 And (HEIS2_ADM_HORIZONTAL_COORD.NORTHING)<=154580.2344) AND ((HEIS2_ADM_HYDRAULIC_HEAD_MV.REVIEW_QUALIFIER) Not Like "F" And (HEIS2_ADM_HYDRAULIC_HEAD_MV.REVIEW_QUALIFIER) Not Like "Y" And (HEIS2_ADM_HYDRAULIC_HEAD_MV.REVIEW_QUALIFIER) Not Like "R")) OR (((HEIS2_ADM_HYDRAULIC_HEAD_MV.HYD_DATE_TIME_PST)>=#1/1/2014#) AND ((HEIS2_ADM_HYDRAULIC_HEAD_MV.HYD_HEAD_METERS_NAVD88) Is Not Null) AND ((HEIS2_ADM_HORIZONTAL_COORD.EASTING)>=571515.1228 And (HEIS2_ADM_HORIZONTAL_COORD.EASTING)<=579316.4375) AND ((HEIS2_ADM_HORIZONTAL_COORD.NORTHING)>=150216.660799999 And (HEIS2_ADM_HORIZONTAL_COORD.NORTHING)<=154580.2344) AND ((HEIS2_ADM_HYDRAULIC_HEAD_MV.REVIEW_QUALIFIER) Is Null))
ORDER BY HEIS2_ADM_HYDRAULIC_HEAD_MV.HYD_DATE_TIME_PST;
