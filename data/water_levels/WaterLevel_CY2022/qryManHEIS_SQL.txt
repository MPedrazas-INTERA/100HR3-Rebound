SELECT HEIS2_ADM_HYDRAULIC_HEAD_MV.WELL_NAME AS NAME, HEIS2_ADM_HYDRAULIC_HEAD_MV.HYD_DATE_TIME_PST AS EVENT, HEIS2_ADM_HYDRAULIC_HEAD_MV.HYD_HEAD_METERS_NAVD88 AS VAL, "MAN" AS TYPE
FROM HEIS2_ADM_HYDRAULIC_HEAD_MV
WHERE (((HEIS2_ADM_HYDRAULIC_HEAD_MV.HYD_HEAD_METERS_NAVD88) Is Not Null) AND ((HEIS2_ADM_HYDRAULIC_HEAD_MV.REVIEW_QUALIFIER) Not Like "F" And (HEIS2_ADM_HYDRAULIC_HEAD_MV.REVIEW_QUALIFIER) Not Like "Y" And (HEIS2_ADM_HYDRAULIC_HEAD_MV.REVIEW_QUALIFIER) Not Like "R")) OR (((HEIS2_ADM_HYDRAULIC_HEAD_MV.HYD_HEAD_METERS_NAVD88) Is Not Null) AND ((HEIS2_ADM_HYDRAULIC_HEAD_MV.REVIEW_QUALIFIER) Is Null))
ORDER BY HEIS2_ADM_HYDRAULIC_HEAD_MV.HYD_DATE_TIME_PST;
